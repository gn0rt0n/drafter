<!-- SQLite DDL conventions (from RESEARCH.md): -->
<!--   bigint PK          → INTEGER PRIMARY KEY AUTOINCREMENT -->
<!--   bigint FK          → INTEGER REFERENCES table(id) -->
<!--   varchar / text     → TEXT -->
<!--   boolean            → INTEGER NOT NULL DEFAULT 0 -->
<!--   enum(a, b, c)      → TEXT NOT NULL DEFAULT 'a' (CHECK constraint optional) -->
<!--   timestamp          → TEXT NOT NULL DEFAULT (datetime('now')) -->
<!--   date               → TEXT -->
<!--   json               → TEXT -->
<!--   UNIQUE(a, b)       → UNIQUE(a, b)  (same) -->
<!-- All migration files: CREATE TABLE IF NOT EXISTS (idempotent) -->
<!-- Every table: id INTEGER PRIMARY KEY AUTOINCREMENT unless composite PK stated -->
<!-- Every entity table: created_at TEXT NOT NULL DEFAULT (datetime('now')), updated_at TEXT NOT NULL DEFAULT (datetime('now')) -->
<!-- FK nullable resolution: acts.start_chapter_id NULLABLE, factions.leader_character_id NULLABLE -->
</interfaces>
