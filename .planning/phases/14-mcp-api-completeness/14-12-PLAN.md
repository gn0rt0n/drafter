Gate-Gated Upsert Pattern for knowledge.py:
```python
@mcp.tool()
async def upsert_reader_reveal(
    reveal_id: int | None,
    chapter_id: int,
    reveal_type: str,
    description: str,
    notes: str | None = None,
) -> GateViolation | ReaderReveal | NotFoundResponse | ValidationFailure:
    async with get_connection() as conn:
        gate = await check_gate(conn)
        if gate:
            return gate
        # Check chapter exists
        ch = await conn.execute_fetchall("SELECT id FROM chapters WHERE id = ?", (chapter_id,))
        if not ch:
            return NotFoundResponse(not_found_message=f"Chapter {chapter_id} not found")
        try:
            if reveal_id is None:
                cursor = await conn.execute(
                    "INSERT INTO reader_reveals (chapter_id, reveal_type, description, notes) VALUES (?, ?, ?, ?)",
                    (chapter_id, reveal_type, description, notes),
                )
                new_id = cursor.lastrowid
                await conn.commit()
                row = await conn.execute_fetchall("SELECT * FROM reader_reveals WHERE id = ?", (new_id,))
            else:
                await conn.execute(
                    "INSERT INTO reader_reveals (id, chapter_id, reveal_type, description, notes) VALUES (?, ?, ?, ?, ?) ON CONFLICT(id) DO UPDATE SET chapter_id=excluded.chapter_id, reveal_type=excluded.reveal_type, description=excluded.description, notes=excluded.notes",
                    (reveal_id, chapter_id, reveal_type, description, notes),
                )
                await conn.commit()
                row = await conn.execute_fetchall("SELECT * FROM reader_reveals WHERE id = ?", (reveal_id,))
            return ReaderReveal(**dict(row[0]))
        except Exception as exc:
            logger.error("upsert_reader_reveal failed: %s", exc)
            return ValidationFailure(is_valid=False, errors=[str(exc)])
```
