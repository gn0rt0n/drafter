Standard two-branch upsert pattern (for upsert_arc):
```python
@mcp.tool()
async def upsert_arc(
    arc_id: int | None,
    character_id: int,
    arc_name: str,
    arc_type: str,
    # ... other fields
) -> CharacterArc | NotFoundResponse | ValidationFailure:
    async with get_connection() as conn:
        # Verify character exists
        char = await conn.execute_fetchall("SELECT id FROM characters WHERE id = ?", (character_id,))
        if not char:
            return NotFoundResponse(not_found_message=f"Character {character_id} not found")
        try:
            if arc_id is None:
                cursor = await conn.execute(
                    "INSERT INTO character_arcs (...) VALUES (...)", (...)
                )
                new_id = cursor.lastrowid
                await conn.commit()
                row = await conn.execute_fetchall("SELECT * FROM character_arcs WHERE id = ?", (new_id,))
            else:
                await conn.execute(
                    "INSERT INTO character_arcs (id, ...) VALUES (?, ...) ON CONFLICT(id) DO UPDATE SET ...",
                    (arc_id, ...)
                )
                await conn.commit()
                row = await conn.execute_fetchall("SELECT * FROM character_arcs WHERE id = ?", (arc_id,))
            return CharacterArc(**dict(row[0]))
        except Exception as exc:
            logger.error("upsert_arc failed: %s", exc)
            return ValidationFailure(is_valid=False, errors=[str(exc)])
```
