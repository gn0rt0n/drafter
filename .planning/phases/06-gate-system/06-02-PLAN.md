```python
"""Shared gate enforcement helper for prose-phase MCP tools.

Placed in novel.mcp (not novel.tools) to avoid circular imports with novel.tools.gate.
Phase 7+ tools (session, timeline, canon, knowledge, foreshadowing, names, voice,
publishing) import check_gate from here and call it at the top of each tool handler.

IMPORTANT: Never use print() in this module. All logging goes to stderr via the
logging module — print() corrupts the stdio protocol.
"""
import logging
import aiosqlite
from novel.models.shared import GateViolation

logger = logging.getLogger(__name__)


async def check_gate(conn: aiosqlite.Connection) -> GateViolation | None:
    """Check whether the architecture gate is certified.

    Call at the top of every prose-phase tool (Phase 7+). If gate is not
    certified, return the GateViolation to the caller immediately.

    Usage in prose-phase tools:
        async with get_connection() as conn:
            violation = await check_gate(conn)
            if violation:
                return violation
            # ... rest of tool logic

    Args:
        conn: An already-open aiosqlite.Connection (from get_connection()).

    Returns:
        GateViolation if gate is not certified (caller must return it).
        None if gate is certified (tool may proceed normally).
    """
    rows = await conn.execute_fetchall(
        "SELECT is_certified FROM architecture_gate WHERE id = 1"
    )
    if not rows or not rows[0]["is_certified"]:
        logger.debug("Gate not certified — returning GateViolation")
        return GateViolation(
            requires_action=(
                "Architecture gate is not certified. Run `novel gate check` to see "
                "what is missing, then `novel gate certify` when all 33 items pass."
            )
        )
    return None
```

This module imports ONLY from `novel.models.shared` and `aiosqlite`. It must NOT import from `novel.tools.*` or `novel.mcp.db` (it receives the connection as a parameter, does not open its own).

def _load_gate_ready(conn: sqlite3.Connection) -> None:
    """Extend minimal seed to satisfy all 33 architecture gate checklist items.

    Calls _load_minimal() first, then adds the rows needed for each gate check.
    All queries are relational ('all X must have Y'), so 3 chapters / 5 characters
    / 6 scenes is sufficient — no need for 55 chapters or 6 POV characters.

    Seed IDs from minimal: characters 1-5, chapters 1-3, scenes 1-6,
    faction id=1 (Obsidian Court), location id=1 (The Ashen Citadel),
    book id=1, act id=1.
    """
    _load_minimal(conn)

    # --- voice_pov: all POV characters (ch1=char1, ch2=char2, ch3=char3) need voice_profiles ---
    # minimal has voice_profile for char 1 only
    for char_id, speech, narrative in [
        (2, "Clipped, tactical, rarely offers more than necessary.",
         "Observational, notices exits and threats first."),
        (3, "Measured and deliberate, always three steps ahead.",
         "Philosophical asides interrupt action beats."),
        (4, "Warm but guarded; uses humour to deflect.",
         "Sensory-rich; food, warmth, texture anchor her scenes."),
        (5, "Formal register with archaic phrasing.",
         "Interior logic exposed in long compound sentences."),
    ]:
        conn.execute(
            "INSERT OR IGNORE INTO voice_profiles "
            "(character_id, speech_patterns, narrative_style, canon_status) "
            "VALUES (?, ?, ?, 'approved')",
            (char_id, speech, narrative),
        )

    # --- rel_pov_pairs: all POV character pairs need character_relationships ---
    # minimal has (1,3),(1,4),(1,5) — add (2,3),(2,4),(2,5),(3,4),(3,5),(4,5)
    for a, b, rel_type in [
        (2, 3, "neutral"), (2, 4, "ally"), (2, 5, "rival"),
        (3, 4, "neutral"), (3, 5, "neutral"), (4, 5, "ally"),
    ]:
        conn.execute(
            "INSERT OR IGNORE INTO character_relationships "
            "(character_a_id, character_b_id, relationship_type, trust_level, conflict_level) "
            "VALUES (?, ?, ?, 3, 2)",
            (min(a, b), max(a, b), rel_type),
        )

    # --- rel_perception: all POV chars need >= 1 perception_profile entry ---
    # minimal has observer=1 -> subject=2; add observers 2,3,4,5
    for observer, subject in [(2, 1), (3, 1), (4, 1), (5, 1)]:
        conn.execute(
            "INSERT OR IGNORE INTO perception_profiles "
            "(observer_id, subject_id, perceived_motivation, trust_assessment) "
            "VALUES (?, ?, 'Unknown', 'Cautious')",
            (observer, subject),
        )

    # --- struct_chapters_hooks: all 3 chapters need opening_hook_note + closing_hook_note ---
    for ch_id, opening, closing in [
        (1, "Open on the moment of refusal — Aeryn denies the summons.",
            "Close on the cost: the letter burns, but the seal is already copied."),
        (2, "Open mid-action — the citadel gate already closing.",
            "Close on a revelation: the informant is family."),
        (3, "Open in silence — the empty council chamber says everything.",
            "Close on a decision that cannot be undone."),
    ]:
        conn.execute(
            "UPDATE chapters SET opening_hook_note=?, closing_hook_note=? WHERE id=?",
            (opening, closing, ch_id),
        )

    # --- struct_chapter_obligations: chapters 2 and 3 need obligations (ch1 already in minimal) ---
    for ch_id, ob_type, desc in [
        (2, "introduce_complication", "Reveal the faction's deeper agenda behind the contract."),
        (3, "resolve_thread", "Close the 'missing seal' subplot before the midpoint break."),
    ]:
        conn.execute(
            "INSERT OR IGNORE INTO chapter_structural_obligations "
            "(chapter_id, obligation_type, description) VALUES (?, ?, ?)",
            (ch_id, ob_type, desc),
        )

    # --- scene_goals: scenes 2-6 need scene_character_goals (scene 1 already in minimal) ---
    for scene_id, char_id, goal in [
        (2, 2, "Obtain the contract seal before Aeryn notices it missing."),
        (3, 1, "Reach the inner vault without triggering the ward sequence."),
        (4, 3, "Delay the council vote by any means necessary."),
        (5, 1, "Extract the informant without breaking cover."),
        (6, 2, "Confirm the seal is genuine — or expose the forgery."),
    ]:
        conn.execute(
            "INSERT OR IGNORE INTO scene_character_goals "
            "(scene_id, character_id, goal, motivation) VALUES (?, ?, ?, 'Survival')",
            (scene_id, char_id, goal),
        )

    # --- pacing_tension + pacing_beats: chapter 3 needs rows (ch1+ch2 already in minimal) ---
    conn.execute(
        "INSERT OR IGNORE INTO tension_measurements "
        "(chapter_id, tension_level, measurement_type) VALUES (3, 7, 'overall')"
    )
    conn.execute(
        "INSERT OR IGNORE INTO pacing_beats "
        "(chapter_id, beat_type, description) VALUES (3, 'climax', 'The council vote and its aftermath.')"
    )

    # --- canon_domains: need >= 3 distinct domains (minimal has 'world' only) ---
    for domain, fact in [
        ("politics", "The Obsidian Court controls all trade licences east of the Vel."),
        ("geography", "The Ashen Wastes cannot be crossed without a Kaelthari guide."),
    ]:
        conn.execute(
            "INSERT OR IGNORE INTO canon_facts "
            "(domain, fact, certainty_level, canon_status) VALUES (?, ?, 'confirmed', 'approved')",
            (domain, fact),
        )

    # --- names_characters + names_locations: all 5 chars + location need name_registry ---
    # minimal has only Aeryn Voss
    for name, entity_type in [
        ("Solvann Drex", "character"),
        ("Ithrel Cass", "character"),
        ("Mira Sundal", "character"),
        ("Calder Veth", "character"),
        ("The Ashen Citadel", "location"),
    ]:
        conn.execute(
            "INSERT OR IGNORE INTO name_registry "
            "(name, entity_type, language_family, canon_status) VALUES (?, ?, 'Vel-tongue', 'approved')",
            (name, entity_type),
        )

    # --- All 33 gate_checklist_items with item_key, category, description ---
    # Use INSERT OR IGNORE — minimal already has 'min_characters' row
    # Import GATE_ITEM_META from tools.gate to avoid duplication
    from novel.tools.gate import GATE_ITEM_META
    for item_key, meta in GATE_ITEM_META.items():
        conn.execute(
            "INSERT OR IGNORE INTO gate_checklist_items "
            "(gate_id, item_key, category, description) VALUES (1, ?, ?, ?)",
            (item_key, meta["category"], meta["description"]),
        )

    conn.commit()
