Existing log_* pattern (from log_character_knowledge in characters.py):
```python
@mcp.tool()
async def log_character_knowledge(
    character_id: int,
    chapter_id: int,
    knowledge_item: str,
    # ... other fields
) -> CharacterKnowledge | NotFoundResponse | ValidationFailure:
    """Log a piece of knowledge gained by a character."""
    async with get_connection() as conn:
        # 1. Check character exists
        char_row = await conn.execute_fetchall(
            "SELECT id FROM characters WHERE id = ?", (character_id,)
        )
        if not char_row:
            return NotFoundResponse(not_found_message=f"Character {character_id} not found")
        # 2. Check chapter exists
        ch_row = await conn.execute_fetchall(
            "SELECT id FROM chapters WHERE id = ?", (chapter_id,)
        )
        if not ch_row:
            return NotFoundResponse(not_found_message=f"Chapter {chapter_id} not found")
        try:
            cursor = await conn.execute(
                "INSERT INTO character_knowledge (...) VALUES (...)",
                (...),
            )
            new_id = cursor.lastrowid
            await conn.commit()
            row = await conn.execute_fetchall(
                "SELECT * FROM character_knowledge WHERE id = ?", (new_id,)
            )
            return CharacterKnowledge(**dict(row[0]))
        except Exception as exc:
            logger.error("log_character_knowledge failed: %s", exc)
            return ValidationFailure(is_valid=False, errors=[str(exc)])
```

get_current helper pattern:
```python
@mcp.tool()
async def get_current_character_location(character_id: int) -> CharacterLocation | NotFoundResponse:
    async with get_connection() as conn:
        rows = await conn.execute_fetchall(
            "SELECT * FROM character_locations WHERE character_id = ? ORDER BY chapter_id DESC LIMIT 1",
            (character_id,)
        )
        if not rows:
            return NotFoundResponse(not_found_message=f"No location found for character {character_id}")
        return CharacterLocation(**dict(rows[0]))
```
