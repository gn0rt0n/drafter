Gate-Gated Upsert Pattern for foreshadowing.py:
```python
@mcp.tool()
async def upsert_motif(
    motif_id: int | None,
    motif_name: str,
    description: str | None = None,
    symbolic_meaning: str | None = None,
    notes: str | None = None,
) -> GateViolation | MotifEntry | ValidationFailure:
    async with get_connection() as conn:
        gate = await check_gate(conn)
        if gate:
            return gate
        try:
            if motif_id is None:
                cursor = await conn.execute(
                    "INSERT INTO motif_registry (motif_name, description, symbolic_meaning, notes) VALUES (?, ?, ?, ?)",
                    (motif_name, description, symbolic_meaning, notes),
                )
                new_id = cursor.lastrowid
                await conn.commit()
                row = await conn.execute_fetchall("SELECT * FROM motif_registry WHERE id = ?", (new_id,))
            else:
                await conn.execute(
                    "INSERT INTO motif_registry (id, motif_name, description, symbolic_meaning, notes) VALUES (?, ?, ?, ?, ?) ON CONFLICT(id) DO UPDATE SET motif_name=excluded.motif_name, description=excluded.description, symbolic_meaning=excluded.symbolic_meaning, notes=excluded.notes",
                    (motif_id, motif_name, description, symbolic_meaning, notes),
                )
                await conn.commit()
                row = await conn.execute_fetchall("SELECT * FROM motif_registry WHERE id = ?", (motif_id,))
            return MotifEntry(**dict(row[0]))
        except Exception as exc:
            logger.error("upsert_motif failed: %s", exc)
            return ValidationFailure(is_valid=False, errors=[str(exc)])
```

Gate-Gated FK-Safe Delete Pattern for foreshadowing.py (delete_motif — motif_registry is referenced by motif_occurrences):
```python
@mcp.tool()
async def delete_motif(motif_id: int) -> GateViolation | NotFoundResponse | ValidationFailure | dict:
    async with get_connection() as conn:
        gate = await check_gate(conn)
        if gate:
            return gate
        row = await conn.execute_fetchall("SELECT id FROM motif_registry WHERE id = ?", (motif_id,))
        if not row:
            return NotFoundResponse(not_found_message=f"Motif {motif_id} not found")
        try:
            await conn.execute("DELETE FROM motif_registry WHERE id = ?", (motif_id,))
            await conn.commit()
            return {"deleted": True, "id": motif_id}
        except Exception as exc:
            logger.error("delete_motif failed: %s", exc)
            return ValidationFailure(is_valid=False, errors=[str(exc)])
```
