FK-Safe Delete Pattern:
```python
@mcp.tool()
async def delete_plot_thread(plot_thread_id: int) -> NotFoundResponse | ValidationFailure | dict:
    async with get_connection() as conn:
        row = await conn.execute_fetchall(
            "SELECT id FROM plot_threads WHERE id = ?", (plot_thread_id,)
        )
        if not row:
            return NotFoundResponse(not_found_message=f"Plot thread {plot_thread_id} not found")
        try:
            await conn.execute("DELETE FROM plot_threads WHERE id = ?", (plot_thread_id,))
            await conn.commit()
            return {"deleted": True, "id": plot_thread_id}
        except Exception as exc:
            logger.error("delete_plot_thread failed: %s", exc)
            return ValidationFailure(is_valid=False, errors=[str(exc)])
```
