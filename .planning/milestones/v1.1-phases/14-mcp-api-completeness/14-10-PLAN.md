Standard two-branch upsert example:
```python
@mcp.tool()
async def upsert_magic_element(
    element_id: int | None,
    name: str,
    element_type: str,
    description: str | None = None,
    rules: str | None = None,
    limitations: str | None = None,
) -> MagicSystemElement | ValidationFailure:
    async with get_connection() as conn:
        try:
            if element_id is None:
                cursor = await conn.execute(
                    "INSERT INTO magic_system_elements (name, element_type, description, rules, limitations) VALUES (?, ?, ?, ?, ?)",
                    (name, element_type, description, rules, limitations),
                )
                new_id = cursor.lastrowid
                await conn.commit()
                row = await conn.execute_fetchall("SELECT * FROM magic_system_elements WHERE id = ?", (new_id,))
            else:
                await conn.execute(
                    "INSERT INTO magic_system_elements (id, name, element_type, description, rules, limitations) VALUES (?, ?, ?, ?, ?, ?) ON CONFLICT(id) DO UPDATE SET name=excluded.name, element_type=excluded.element_type, description=excluded.description, rules=excluded.rules, limitations=excluded.limitations",
                    (element_id, name, element_type, description, rules, limitations),
                )
                await conn.commit()
                row = await conn.execute_fetchall("SELECT * FROM magic_system_elements WHERE id = ?", (element_id,))
            return MagicSystemElement(**dict(row[0]))
        except Exception as exc:
            logger.error("upsert_magic_element failed: %s", exc)
            return ValidationFailure(is_valid=False, errors=[str(exc)])
```
