Gate-Gated Delete Pattern (for voice.py and session.py):
```python
@mcp.tool()
async def delete_voice_profile(voice_profile_id: int) -> GateViolation | NotFoundResponse | ValidationFailure | dict:
    async with get_connection() as conn:
        gate = await check_gate(conn)
        if gate:
            return gate
        row = await conn.execute_fetchall(
            "SELECT id FROM voice_profiles WHERE id = ?", (voice_profile_id,)
        )
        if not row:
            return NotFoundResponse(not_found_message=f"Voice profile {voice_profile_id} not found")
        try:
            await conn.execute("DELETE FROM voice_profiles WHERE id = ?", (voice_profile_id,))
            await conn.commit()
            return {"deleted": True, "id": voice_profile_id}
        except Exception as exc:
            logger.error("delete_voice_profile failed: %s", exc)
            return ValidationFailure(is_valid=False, errors=[str(exc)])
```
